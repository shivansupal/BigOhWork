#include<bits/stdc++.h>
using namespace std;


void printarray(int arr[],int n){
    
    sort(arr,arr+n);
    
    int temp[n];
    int j=0;
    
    for(int i=1;i<=n;i++){
        if(arr[i]!=arr[i-1]){
            
            temp[j++]=arr[i-1];
        }
        
        
        
    }
    
   
    
    
    for(int i=0;i<j;i++){
        
        cout<<temp[i];
    }
}

int main(){
    
    int n;
    cin>>n;
    
    int arr[n];
    for(int i=0;i<n;i++){
        
        cin>>arr[i];
    }
    
    printarray(arr,n);
    
  /*for(int i=0;i<n;i++){

    return 0;
    
    
    
}
