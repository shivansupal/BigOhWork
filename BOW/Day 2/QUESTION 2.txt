#include<bits/stdc++.h>

using namespace std;

int main()
{
    
    int n;
    cout<<"Enter the value of N"<<endl;
    cin>>n;
    cout<<"Your pattern is:"<<endl<<endl;
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            if(i==j){
                
                cout<<"\\";
            }
            
            else if(j==n-i-1){
                
                cout<<"/";
            }
            
            else{
                
                cout<<"*";
    }
            
        }
    cout<<endl;
    }
     return 0;
}
